	\vspace{-0.25cm}
	\begin{block}
		\scriptsize Dans les réseaux de capteurs sans fil, le regroupement des n?uds de capteurs en
Les petits groupes sont une technique efficace pour atteindre l'extensibilité,
l'auto-organisation, l'économie d'énergie, l'accès aux canaux, le routage, etc.

protocoles de formation de clusters peuvent être regroupé en deux grandes catégories:

	\end{block}
	\end{frame}
	
	\begin{frame}{Contexte }
					\transwipe
					\vspace{-0.21cm}
					\begin{block}{}
							\begin{itemize}
								\item[\ding{43}] Lead-First:
								\begin{itemize}
										\item Dans l'approche lead-first, les clusters head sont d'abord élus selon certaines métriques (batteries restantes, degré de connectivité). 
										\item Ensuite, les clusters heads s'accordent sur comment assigner les noeuds à différents clusters.
									\end{itemize}
								\item[\ding{43}]Cluster-first
 \begin{itemize}
										\item Dans cette approche, tous les n?uds sont d'abord regroupés en clusters,
										\item  Et ensuite chaque cluster sélectionné son cluster head sur la coordination de tous les n?uds du cluster.
									\end{itemize}
							\end{itemize}
					
					\end{block}
				\end{frame}

	\transwipe

\section{ Secure distributed cluster formation in wireless sensor networks}	
			\begin{frame}{}
				\transfade
				\begin{center}
					\vspace{-0.2cm}
				\huge\textbf{ \textsc{ secure distributed cluster formation in wireless sensor networks}}
				\end{center}
			\end{frame}
			
			
			\subsection{Définitions}
			\begin{frame}{Définitions}
				\transwipe
				\vspace{-0.21cm}
					\begin{block}{}
						\begin{itemize}
							\item[\ding{43}] Le protocole propose ici est: un protocole sécurisé de formation de cluster distribuées pour organiser les réseaux de capteurs en cliques mutuellement disjointes
							\item[\ding{43}]Clique: est cluster dans lequel chaque noeud peut communiquer avec tous ses voisins à un saut.
						\end{itemize}
					\end{block}
					
			\end{frame}
			
			
			
			
			
			
			
			\subsection{Propriétés}
			\begin{frame}{Propriétés}
				\transwipe
				\vspace{-0.21cm}
					\begin{block}{\scriptsize Le protocole de formation de cluster distribué sécurisé possède les propriétés suivantes même s'il y a des attaquants externes et internes}
						\begin{itemize}
							\item[\ding{43}]Le protocole est entièrement distribué. Chaque noeud calcule sa clique uniquement en utilisant les informations de ses voisins;
							\item[\ding{43}] La fin du protocole est garantie. Les noeuds participants qui ne respectent pas les spécifications du protocole (p. ex., envoyer des messages contradictoires) seront identifiés et retirés de toutes les cliques;
						\end{itemize}
					\end{block}
					
			\end{frame}
			\begin{frame}{Propriétés}
				\transwipe
				\vspace{-0.21cm}
					\begin{block}{\scriptsize Le protocole de formation de cluster distribué sécurisé possède les propriétés suivantes même s'il y a des attaquants externes et internes}
						\begin{itemize}
						\item[\ding{43}] Une fois le protocole terminé,
						\begin{itemize}
										\item Tous les noeuds normaux sont divisés en cliques disjointes.
										\item  Tous les noeuds normaux sont garantis d'avoir des vues cohérentes sur leurs adhésions à la clique, même dans un environnement hostile;

									\end{itemize}
						\end{itemize}
					\end{block}
					
			\end{frame}
			\begin{frame}{propriétés}
				\transwipe
					\vspace{-0.21cm}
						\begin{block}{\scriptsize Le protocole de formation de cluster distribué sécurisé possède les propriétés suivantes même s'il y a des attaquants externes et internes}
							\begin{itemize}
								\item[\ding{43}]les attaquants internes qui ne suivent pas la sémantique du protocole peuvent étre identifiés et retirés du réseau ;
								\item[\ding{43}] les attaquants externes peuvent être empêchés de participer au processus de formation du cluster;
								
								\item[\ding{43}]  les coûts de communication sont modéré
							\end{itemize}
						\end{block}
			\end{frame}
			\subsection{Hypothèses }
				\begin{frame}{Hypothèses }
					\transwipe
					\vspace{-0.21cm}
					\begin{block}{}
							\begin{itemize}
								\item[\ding{43}]Chaque 	noeud connaît ses voisins 1-hop
								\item[\ding{43}]Les noeuds de capteurs peuvent effectuer des opérations de signature numérique à clé publique.

								\item[\ding{43}]Les horloges des noeuds normaux sont synchronisées de manière lâche, comme l'exige uTESLA.
								\item[\ding{43}]Les clés publiques utilisées par les noeuds capteurs sont correctement authentifiées
 
							\end{itemize}
					
					\end{block}
				\end{frame}
				
				\subsection{Objectif}
				\begin{frame}{Objectif}
					\transwipe
					\vspace{-0.21cm}
					\begin{block}{}
							L'objectif premier de ce protocole est de diviser tous les noeuds du réseau en des cliques mutuelles disjoints de telle sorte que chaque noeud dans un même clique puisse communiquer directement avec chacun de ses voisins
					\end{block}
				\end{frame}
				
				
				
			\subsection{Spécification du protocole}
			
			\begin{frame}{Spécification du protocole}
					\transwipe
					\vspace{-0.21cm}
					On définit par:
					\begin{itemize}
								\item C$_i$ = Clique local du noeud i.
								\item Clique Agreement: pour chaque noeud j $\in$ C$_i$,   C$_j$ = C$_i$
								\item Clique inconsistency:  il existe j $\in$ C$_i$ tel que C$_j$ $\neq$ C$_i$
 
							\end{itemize}
					
				\end{frame}			
				\begin{frame}{Spécification du protocole}
					\transwipe
					\vspace{-0.21cm}
					\begin{figure}
						\begin{center}
							\includegraphics[width=11cm,height=7cm]{images/heur.PNG}
						\end{center}
					\end{figure}
					
				\end{frame}		
				
				\begin{frame}{Spécification du protocole}
					\transwipe
					\vspace{-0.21cm}
					\begin{itemize}
								\item L$_i$ = liste des voisins de i à 1 saut
								\item V$_i$ = $\lbrace i \rbrace +  $ la liste de voisins de i
								\item $E_i = $ ensemble des bidirections entre les noeuds dans V$_i$
								\item $C_i^k = $ Clique de i a l'étape k, avec $ 1 \leq k \leq 4 $
							\end{itemize}
					
				\end{frame}	
				
				
				\begin{frame}{Spécification du protocole}
					\transwipe
					\vspace{-0.21cm}
					Exemple:
					\begin{figure}
						\begin{center}
							\includegraphics[width=11cm,height=6cm]{images/exemple.PNG}
						\end{center}
					\end{figure}
					
				\end{frame}	
				
\begin{frame}{Spécification du protocole}
					\transwipe
					\vspace{-0.21cm}
					 Etape 1: Chaque noeud échange ses listes de voisins avec ses voisins et calcule sa clique maximale locale.
					
				\end{frame}		
				\begin{frame}{Spécification du protocole}
					\transwipe
					\vspace{-0.21cm}
					 $S_0 = \lbrace 1,2 \rbrace$, $L_0 = \lbrace 1,2\rbrace$, $C_0 = \lbrace 0 \rbrace$ \\
					 $L_1 = \lbrace 0,2,3,4,7 \rbrace$, $\vert L_0\cap L_1 \vert= \vert\lbrace 2 \rbrace\vert = 1$ \\
					 $L_2 = \lbrace 0,1,3\rbrace$, $\vert L_0\cap L_2 \vert= \vert\lbrace 1 \rbrace\vert = 1$  \\
					 comme les noeuds 1 et 2 ont le meme monbre d'element on choisie le noeud  1 \\
					 
					 $L_0 = \lbrace2\rbrace $,   $C_0 = \lbrace 0,1 \rbrace$,   $S_0 = \lbrace 2 \rbrace$ \\
					 $\vert L_0\cap L_2 \vert= \vert\lbrace 1 \rbrace\vert = 1$ \\ 
					 $L_0 = \Phi$,$C_0 = \lbrace 0,1,2 \rbrace$,   $S_0 = \Phi$ \\ 
					 
					  $C_0^1 = \lbrace 0,1,2 \rbrace $, De même, nous avons\\
					 $C_0^1 = C_1^1 = C_2^1  \lbrace 0,1,2 \rbrace $,
					 $C_3^1 = C_4^1 = C_5^1 = C_6^1 = \lbrace 3,4,5,6  \rbrace $,$C_7^1 = \lbrace 1,7\rbrace $,
					 
				\end{frame}		
				
	\begin{frame}{Spécification du protocole}
					\transwipe
					\vspace{-0.21cm}
					 Etape 2: Chaque noeud:
									\begin{itemize}
										\item échange sa clique maximale locale avec ses voisins,
										\item après avoir reçu les cliques maximales locales de ses voisins, le noeud i vérifie s'il existe une  clique $C_j^1$ qui soit "meilleure" que sa clique $C_i^1$
								
									\item si le noeud i recois n clique contenant i,il ordonne les cliques comme\\ $C^1_{\alpha 1} \prec^i \ldots \prec^i C^1_{\alpha i} \prec^i \ldots \prec^i C^1_{\alpha n}$, 
									$C^2_i = C^1_{\alpha n}$
									\end{itemize}
									
									Pour comparer les cliques calculées par différents noeuds, nous utilisons cette relation $\prec^i$ sur les cliques comme suit
									
					
				\end{frame}		
				
				\begin{frame}{Spécification du protocole}
					\transwipe
					\vspace{-0.21cm}
					\begin{figure}
						\begin{center}
							\includegraphics[width=11cm,height=6cm]{images/order.PNG}
						\end{center}
					\end{figure}
					
				\end{frame}	
				
			\begin{frame}{Spécification du protocole}
					\transwipe
					\vspace{-0.21cm}
					 $C_0^1 = C_1^1 = C_2^1 = \lbrace 0,1,2 \rbrace $,
					 $C_3^1 = C_4^1 = C_5^1 = C_6^1 = \lbrace 3,4,5,6  \rbrace $,$C_7^1 = \lbrace 1,7\rbrace $,
					 
					 Noeud 1:\\
					 $\vert C_7^1\vert < \vert C_0^1\vert$, $C^1_7 \prec^i C^1_0$ \\
					 $\vert C_0^1\vert = \vert C_1^1\vert = \vert C_1^2\vert$ et  $0<1<2$ donc on a : \\
					 $C^1_7 \prec^1 C^1_0 \prec^1 C^1_1 \prec^1 C^1_2$ Donc on a:  $C^2_1 =  C^1_2 =  \lbrace 0,1,2 \rbrace $ \\  De même, nous avons\\
					  $C_0^2 = C_1^2 = C_2^2  \lbrace 0,1,2 \rbrace $,
					 $C_3^2 = C_4^2 = C_5^2 = C_6^2 = \lbrace 3,4,5,6  \rbrace $,$C_7^2 = \lbrace 1,7\rbrace $,
					 					 
				\end{frame}					
			\begin{frame}{Spécification du protocole}
					\transwipe
					\vspace{-0.21cm}Etape 3: Chaque noeud: 
										\begin{itemize}
										\item échange la clique mise à jour avec ses voisins
										\item Pour chaque noeud j dans $C_2^1$, le noeud i vérifie s'il est inclus dans j
clique $C_2^j$ Sinon, le noeud i retire j de sa clique $C_2^i$ .	
										\item Si le noeud i ne reçoit pas la clique actualisée du noeud j, le n?ud i garde simplement le n?ud j dans sa clique.

									\end{itemize}
									Exemple:  $C_1^2  = \lbrace 0,1,2 \rbrace $, $C_7^2 = \lbrace 1,7\rbrace $\\
									$C_2^1$ ne contient pas 7 donc le noeud 7 retire 1 de sa clique et on a : $C_7^3 = \lbrace7\rbrace$
				\end{frame}					
				
				\begin{frame}{Spécification du protocole}
					\transwipe
					\vspace{-0.21cm}Etape 4: Chaque noeud échange sa  clique final avec ses voisins.
								
									\begin{itemize}
										\item  Si $ j \in C_i^3$ , node i rediffuse la clique $C_j^3$ . L'objectif de cette rediffusion est de prévenir les attaques silencieuses. 
										\item Chaque Noeud i vérifie que, pourtous $ j \in C_i^3$,  $C_j^3 = C_i^3$ 
										\item Si aucune incohérence de clique n'est détectée, il se termine avec succès.
										\item
Sinon, il entre à l'étape 5.
									\end{itemize}
				\end{frame}			
				
				\begin{frame}{Spécification du protocole}
					\transwipe
					\vspace{-0.21cm}étape 5: Chaque noeud effectue un contrôle de conformité.\\ 
					Cette étape se déroule en deux temps. Dans la première étape, 
									\begin{itemize}
									
										\item Conformity Checking: le noeud i effectue le contrôle de conformité pour identifier les noeuds malveillants qui ont envoyé des messages incohérents au cours des quatre étapes précédentes
										\item  Consistency Enforcement: permet de faire respecter l'accord de clique, et terminer le protocole de formation de la clique.
									\end{itemize}									 

				\end{frame}
